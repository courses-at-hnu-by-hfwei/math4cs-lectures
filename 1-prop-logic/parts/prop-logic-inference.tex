\begin{frame}{}
  % Example 1.4
  % Sequent: p \land q, r \vdash q \land r
  \begin{exampleblock}{``$\land$''推理规则的应用}
    \[
      \set{p \land q, r} \vdash q \land r
    \]
  \end{exampleblock}

  \vspace{0.50cm}
  \begin{proof}
    \begin{logicproof}{1}
      \uncover<2->{p \land q} & \uncover<2->{premise} \\
      \uncover<2->{r} & \uncover<2->{premise} \\
      \uncover<4->{q} & \uncover<4->{$\land \elimright$ 1} \\
      \uncover<3->{q \land r} & \uncover<5->{$\land \intro$ 3,2}
    \end{logicproof}
  \end{proof}
\end{frame}

  \begin{definition}[Contradiction ($\bot$; Falsum, Falsehood)]
    $\bot$ is a special formula of the form
    $\alpha \land \lnot \alpha$ or $\lnot \alpha \land \alpha$,
    where $\alpha$ is any formula.
  \end{definition}

\begin{frame}{}
  % Example 1.20
  % Sequent: \neg p \lor q \vdash p \to q
  % Normalized from the source's side-by-side case layout.
  \begin{exampleblock}{``$\bot$''-推理规则的应用}
    \[
      \lnot p \lor q \vdash p \to q
    \]
  \end{exampleblock}

  {\scriptsize
  \begin{proof}
    \begin{logicproof}{3}
      \uncover<2->{\neg p \lor q} & \uncover<2->{premise} \\
      \uncover<4->{\begin{subproof}
        \uncover<4->{\lnot p} & \uncover<4->{premise} \\
      \uncover<6->{\begin{subproof}
      \uncover<6->{p} & \uncover<6->{assumption} \\
      \uncover<7->{\bot} & \uncover<7->{$\lnot\elim$ 3, 2} \\
      \uncover<8->{q} & \uncover<8->{$\bot\elim$ 4}
      \end{subproof}}
      \uncover<5->{p \to q} & \uncover<9->{$\to\intro$ 3--5}
      \end{subproof}}
      \uncover<10->{\begin{subproof}
      \uncover<10->{q} & \uncover<10->{premise} \\
      \uncover<12->{\begin{subproof}
      \uncover<12->{p} & \uncover<12->{assumption} \\
      \uncover<13->{q} & \uncover<13->{copy\ 7}
      \end{subproof}}
      \uncover<11->{p \to q} & \uncover<14->{$\to\intro$ 8--9}
      \end{subproof}}
      \uncover<3->{p \to q} & \uncover<15->{$\lor\elim$ 1, 2--6, 7--10}
    \end{logicproof}
  \end{proof}}
\end{frame}

\begin{frame}{}
  % Example 1.21(a)
  % Sequent: p \to q, p \to \lnot q \vdash \lnot p
  \begin{exampleblock}{``$\lnot$''-推理规则的应用}
    \[
      \set{p \to q, p \to \lnot q} \vdash \lnot p
    \]
  \end{exampleblock}

  \vspace{0.50cm}
  \begin{proof}
    \begin{logicproof}{1}
      \uncover<2->{p \to q} & \uncover<2->{premise} \\
      \uncover<2->{p \to \lnot q} & \uncover<2->{premise} \\
      \uncover<4->{\begin{subproof}
        \uncover<4->{p} & \uncover<4->{assumption} \\
        \uncover<6->{q} & \uncover<6->{$\to\elim$ 1, 3} \\
        \uncover<7->{\lnot q} & \uncover<7->{$\to\elim$ 2, 3} \\
        \uncover<5->{\bot} & \uncover<8->{$\lnot\elim$ 4, 5}
      \end{subproof}}
      \uncover<3->{\lnot p} & \uncover<9->{$\lnot\intro$ 3--6}
    \end{logicproof}
  \end{proof}
\end{frame}

      \uncover<4->{\begin{subproof}

        \uncover<7->{r} & \uncover<7->{$\to\elim$ 5, 4} \\

      \end{subproof}}

    \end{logicproof}
  \end{proof}

\begin{frame}{}
  % Example 1.23
  % Sequent: p \land \lnot q \to r, \lnot r, p \vdash q
  \begin{exampleblock}{``$\lnot$''-推理规则的应用}
    \[
      \set{(p \land \lnot q) \to r, \lnot r, p} \vdash q
    \]
  \end{exampleblock}

  \vspace{0.50cm}
  {\small
  \begin{proof}
    \begin{logicproof}{1}
      \uncover<2->{(p \land \lnot q) \to r} & \uncover<2->{premise} \\
      \uncover<2->{\lnot r} & \uncover<2->{premise} \\
      \uncover<2->{p} & \uncover<2->{premise} \\
      \uncover<5->{\begin{subproof}
        \uncover<5->{\lnot q} & \uncover<5->{assumption} \\
        \uncover<7->{p \land \lnot q} & \uncover<7->{$\land\intro$ 3, 4} \\
        \uncover<8->{r} & \uncover<8->{$\to\elim$ 1, 5} \\
        \uncover<6->{\bot} & \uncover<9->{$\lnot\elim$ 6, 2}
      \end{subproof}}
      \uncover<4->{\lnot\lnot q} & \uncover<10->{$\lnot\intro$ 4--7} \\
      \uncover<3->{q} & \uncover<11->{$\lnot\lnot\elim$ 8}
    \end{logicproof}
  \end{proof}}
\end{frame}

  \begin{exampleblock}{使用命题逻辑进行推理}
    某公司要从赵、钱、孙、李、吴5名员工中选派某些人出国考察。\\
    由于某些不可描述的原因, 选派要求如下:

    \begin{columns}
      \column{0.50\textwidth}
        \begin{enumerate}[(1)]
          \item 若赵去, 钱也去;
          \item 李、吴两人中必有一人去;
          \item 钱、孙两人中去且仅去一人;
          \item 孙、李两人同去或同不去;
          \item 若吴去, 则赵、钱也去;
          \item 只有孙去, 赵才会去。
        \end{enumerate}
      \column{0.50\textwidth}
        \begin{enumerate}[(1)]
          \item $\teal{Z \to Q}$;
          \item $\teal{L \lor W}$;
          \item $\teal{(Q \land \lnot S) \lor (S \land \lnot Q)}$;
          \item $\teal{(S \land L) \lor (\lnot S \land \lnot L)}$;
          \item $\teal{W \to Z \land Q}$;
          \item $\teal{Z \to S}$。
        \end{enumerate}
    \end{columns}

    \vspace{0.50cm}
    \blue{请使用形式化推理的方法帮该公司判断应选哪些人出国考察。}
  \end{exampleblock}
\end{frame}
