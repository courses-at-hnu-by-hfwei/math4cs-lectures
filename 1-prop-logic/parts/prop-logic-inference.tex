\begin{frame}{}
  % Example 1.4
  % Sequent: p \land q, r \vdash q \land r
  \begin{exampleblock}{``$\land$''推理规则的应用}
    \[
      \set{p \land q, r} \vdash q \land r
    \]
  \end{exampleblock}

  \pause
  \vspace{0.50cm}
  \begin{proof}
    \begin{logicproof}{1}
      p \land q & premise \\
      r & premise \\
      q & $\land \elimright$ 1 \\
      q \land r & $\land \intro$ 3,2
    \end{logicproof}

    % \begin{align}
    %   &p \land q &\text{前提} \label{eqn:pq} \\
    %   &r &\text{前提} \label{eqn:r} \\
    %   &q &\land{\text{-elim-right}\; (\ref{eqn:pq}}) \label{eqn:q} \\
    %   &q \land r &\land{\text{-intro}}\; (\ref{eqn:q}), (\ref{eqn:r})
    % \end{align}
  \end{proof}
\end{frame}

\begin{frame}{}
  % Example 1.5
  % Sequent: p, \neg\neg(q \land r) \vdash \neg\neg p \land r
  \begin{exampleblock}{``$\land$''推理规则的应用}

      \set{p, \lnot\lnot(q \land r)} \vdash \lnot\lnot p \land r

  \end{exampleblock}

  \pause

    \begin{logicproof}{1}

      q \land r & $\lnot\lnot \elim$ 2 \\
      r & $\land \elimright$ 4 \\
      \lnot\lnot p \land r & $\land \intro$ 3,5
    \end{logicproof}
  \end{proof}
\end{frame}

\begin{frame}{}
  % Example 1.11
  % Sequent: \vdash (q \to r) \to ((\lnot q \to \lnot p) \to (p \to r))
  \begin{exampleblock}{``$\to$''推理规则的应用}

      \vdash (q \to r) \to ((\lnot q \to \lnot p) \to (p \to r))

            q & $\lnot\lnot\elim$ 5 \\
            r & $\to\elim$ 1, 6
          \end{subproof}
          p \to r & $\to\intro$ 3--7
        \end{subproof}
        (\lnot q \to \lnot p) \to (p \to r) & $\to\intro$ 2--8
      \end{subproof}
      (q \to r) \to ((\lnot q \to \lnot p) \to (p \to r)) & $\to\intro$ 1--9
    \end{logicproof}
  \end{proof}

          \item $\teal{(S \land L) \lor (\lnot S \land \lnot L)}$;
          \item $\teal{W \to Z \land Q}$;
          \item $\teal{Z \to S}$。
        \end{enumerate}
    \end{columns}

  \begin{theorem}[命题逻辑的可靠性 (Soundness)]
    如果 $\Sigma \vdash \alpha$, 则 $\Sigma \models \alpha$。
  \end{theorem}

  \begin{theorem}[命题逻辑的完备性 (Completeness)]

  \end{theorem}
