\begin{frame}{}
  \begin{definition}[可比与不可比 (Comparable/Incomparable)]
    设 $(X, \preceq)$ 是偏序集。
    对 $a, b \in X$,

    \begin{description}
      \item[可比的 (Comparable):]
        \[
          a \preceq b \lor b \preceq a
        \]
      \item[不可比的 (Incomparable):]
        \[
          \lnot(a \preceq b \lor b \preceq a)
        \]
    \end{description}
  \end{definition}

  \fig{width = 0.50\textwidth}{figs/Hasse-Powerset}
\end{frame}

\begin{frame}{}
  \begin{definition}[链与反链 (Chain; Antichain)]
    设 $(X, \preceq)$ 是偏序集。

    \begin{itemize}
      \setlength{\itemsep}{6pt}
      \item 设 $S \subseteq X$ 且 $S$ 中元素两两\blue{可比}, 则称 $S$ 是\red{链}。
      \item 设 $S \subseteq X$ 且 $S$ 中元素两两\blue{不可比}, 则称 $S$ 是\red{反链}。
    \end{itemize}
  \end{definition}

  \begin{center}
    注意: 单元素集合既是链, 也是反链
  \end{center}

  \fig{width = 0.50\textwidth}{figs/Hasse-Powerset}
\end{frame}

\begin{frame}{}
  \begin{definition}[宽度与高度 (Width; Height)]
    设 $(X, \preceq)$ 是偏序集。

    \begin{itemize}
      \setlength{\itemsep}{6pt}
      \item \red{宽度 (Width)}: $X$ 中最大反链的大小
      \item \red{高度 (Height)}: $X$ 中最大链的大小
    \end{itemize}
  \end{definition}

\begin{frame}{}
  \begin{definition}[链分解 (Chain Decomposition)]
    设 $(X, \preceq)$ 是偏序集。
    对于 $C_{1}, C_{2}, \dots, C_{k} \subseteq X$, 如果
    \begin{enumerate}
      \setlength{\itemsep}{6pt}
      \item 每个 $C_{i}$ 都是\red{\bf 链};
      \item $C_{1}, C_{2}, \dots, C_{k}$ \red{\bf 互不相交};
      \item $X = \bigcup_{i=1}^{k} C_{i}$,
    \end{enumerate}
    则称 $\set{C_{1}, C_{2}, \dots, C_{k}}$
    是 $X$ 的\blue{\bf 链分解}。
  \end{definition}

  \fig{width = 0.40\textwidth}{figs/Hasse-Powerset}
\end{frame}

\begin{frame}{}
  \begin{lemma}[$\le$: Upper Bound of Chain Decomposition Size]
    \[
      \text{minimum chain-decomposition size} \le \operatorname{width}(X).
    \]
  \end{lemma}

  \begin{proof}
    \pause
    \blue{\bf By induction on \pause the size $|X|$ of the poset}

    \pause
    \vspace{1em}
    \begin{description}[Inductive Hypothesis:]
      \setlength{\itemsep}{8pt}
      \item[Base Case:] $|X| = 1$; It holds obviously.
      \pause
      \item[Inductive Hypothesis:]
        Suppose the theorem holds for all posets
        of size $2 \le |X| < n$.
      \pause
      \item[Inductive Step:]
        Consider a poset $X$ of size $|X| = n$.
        \vspace{0.20cm}
        \begin{itemize}
          \setlength{\itemsep}{6pt}
          \pause
          \item Suppose $A$ is a maximum antichain of $X$ with $k$ elements.
          \pause
          \item We construct a chain decomposition of $X$ with $k$ chains.
        \end{itemize}
    \end{description}
  \end{proof}
\end{frame}

\begin{frame}{}
  \begin{lemma}[$\le$: Upper Bound of Chain Decomposition Size]
    \[
      \text{minimum chain-decomposition size} \le \operatorname{width}(X).
    \]
  \end{lemma}

  \begin{proof}
    \begin{description}
      \setlength{\itemsep}{8pt}
      \item[Case 1:]
        Every largest antichain is
        exactly the set of all minimal elements,
        or exactly the set of all maximal elements.
      \item[Case 2:]
        There exists a largest antichain
        that is neither the set of all minimal elements
        nor the set of all maximal elements.
    \end{description}
  \end{proof}
\end{frame}

\begin{frame}{}
  \begin{lemma}[$\le$: Upper Bound of Chain Decomposition Size]
    \[
      \text{minimum chain-decomposition size} \le \operatorname{width}(X).
    \]
  \end{lemma}

  \begin{proof}
    \begin{center}
      \fbox{
        Case 1: Every largest antichain is
        \blue{\bf exactly the set of all minimal elements},
        or exactly the set of all maximal elements.
      }

      \pause
      \vspace{0.30cm}
      Let $M$ be the largest antichain of $X$,
      consisting of exactly the set of all minimal elements of $X$.

      \pause
      \vspace{0.30cm}
      Choose some minimal element $m \in M$.

      \pause
      \vspace{0.30cm}
      Choose a maximal chain $C^{m}$ starting from $m$:
      \[
        C^{m}: m = x_{1} \prec x_{2} \prec \cdots \prec x_{t}
      \]

      \pause
      \vspace{0.30cm}
      $x_{t}$ is a maximal element of $X$

      \pause
      \vspace{0.30cm}
      Remove $C^{m}$ from $X$ to get a smaller poset $X' = X \setminus C^{m}$.

      \pause
      \fbox{$\operatorname{Width}(X') \le n - 1$}

      \pause
      \vspace{0.30cm}
      By the inductive hypothesis,
      $X'$ can be decomposed into $\le (n - 1)$ chains.

      \pause
      \vspace{0.30cm}
      Adding $C^{m}$ back gives a chain decomposition
      of $X$ with $\le n$ chains.
    \end{center}
  \end{proof}
\end{frame}
