\begin{frame}{}
  \begin{definition}
    设 $(X, \preceq)$ 是偏序集。
    对任意 $a, b \in X$, 如果
    \[
      a \prec b \land
        (\forall c \in X\; (c \neq a \land c \neq b)
          \to \lnot (a \preceq c \preceq b))
    \]
    则称 $a$ 被 $b$ 覆盖 (covered)。
  \end{definition}

  \begin{center}
    \fig{width = 0.40\textwidth}{figs/Hasse-Powerset}
    哈斯图 (Hasse diagram)
  \end{center}
\end{frame}
