\begin{frame}{}
  \begin{definition}
    设 $(X, \preceq)$ 是偏序集。
    对任意 $a, b \in X$,

    \begin{description}
      \item[严格小于 (strictly less than):]
        \[
          a \prec b \triangleq a \preceq b \land a \neq b
        \]
      \item[$a$被$b$覆盖 (covered by):]
        \[
          a \prec b \land
            (\forall c \in X.\; (c \neq a \land c \neq b) \to \lnot (a \preceq c \preceq b))
        \]
    \end{description}
  \end{definition}

\end{frame}
