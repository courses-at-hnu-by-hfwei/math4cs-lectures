\begin{frame}{}
  \begin{theorem}[一阶谓词逻辑的可靠性 (Soundness)]
    如果 $\Sigma \vdash \alpha$, 则 $\Sigma \models \alpha$。
  \end{theorem}

  \vspace{0.30cm}
  \fig{width = 0.50\textwidth}{figs/syntax-semantics}
  \vspace{0.30cm}

  \begin{theorem}[一阶谓词逻辑的完备性 (Completeness)]
    如果 $\Sigma \models \alpha$, 则 $\Sigma \vdash \alpha$。
  \end{theorem}
\end{frame}
